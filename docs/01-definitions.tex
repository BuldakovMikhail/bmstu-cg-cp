\begin{definitions}
	\definition{Яркость}{световая величина $L$, характеризующая излучательную способность источника света~\cite{encycle}.}
	\definition{Луч}{понятие геометрической оптики, обозначающие линию, вдоль которой распространяется поток энергии, испущенной источником света~\cite{encycle}.}
	\definition{Интенсивность излучения}{энергетическая характеристика распространяющегося в заданном направлении электромагнитного излучения, пропорциональная квадрату амплитуды его волны~\cite{encycle}.}
	\definition{Длина волны}{расстояние между двумя ближайшими точками гармонической бегущей волны, которые находятся в одинаковой фазе~\cite{encycle}.}
	\definition{Плотность}{масса, приходящаяся на единицу объема однородного вещества~\cite{encycle}.}
\end{definitions}
